CREATE TABLE usuario(

	id serial not null,
	nome varchar(50),
	email varchar(50),
	senha varchar(50),
	CONSTRAINT "usuario_pk" PRIMARY KEY("id")
);
CREATE TABLE feedDeNoticia(

	id serial not null,
	localidade varchar(80),
	hora Timestamp,
	descricao text,
	categoria varchar(50),
	id_usuario integer,
	CONSTRAINT "feedDeNoticia_pk" PRIMARY KEY("id"),
	CONSTRAINT feeds_fk FOREIGN KEY (id_usuario) REFERENCES usuario (id)
);

CREATE TABLE orgao(

	id serial not null,
	nome varchar(200),
	senha varchar(150),
	CONSTRAINT "orgao_pk" PRIMARY KEY("id")
);
